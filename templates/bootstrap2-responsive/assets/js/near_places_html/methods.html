Methods

Interface with bootstrap-select.

.near_places('setMarker')

You can set marker or change.

$('#near_places').near_places('setMarker', {ltng :[45.812231, 15.920618], 'address':'Ilica 345, HR-10000'});

.near_places('setZoom')

This will set Zoom of map.

$('.near_places').near_places('setZoom', 5);

.near_places('setCenter')

This will be set Center of map.

$('.near_places').near_places('setCenter', [45.812231, 15.920618]);

.near_places('setCenterByMarker')

This will be center map by Marker.

$('.near_places').near_places('setCenterByMarker');

.near_places('clearPlaces')

This will be clear all markers, places icons from map

$('.near_places').near_places('clearPlaces');

.near_places('destroy')

This will be destroy map

$('.near_places').near_places('destroy');
